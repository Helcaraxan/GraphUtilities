include(ExternalProject)

ExternalProject_Add(
	mt-metis
	TMP_DIR ${CMAKE_CURRENT_BINARY_DIR}/tmp
	STAMP_DIR ${CMAKE_CURRENT_BINARY_DIR}/stamp
	URL ${CMAKE_CURRENT_SOURCE_DIR}/mt-metis-0.3.tar.gz
	#	URL_MD5 b430f097a53435d2ca4cf4587eb45159
	URL_MD5 54f87ba549787b23a1bc58b4b14aceab
	PREFIX ${CMAKE_CURRENT_BINARY_DIR}/mt-metis-0.3.0
	SOURCE_DIR ${CMAKE_CURRENT_BINARY_DIR}/mt-metis-0.3.0
	BUILD_IN_SOURCE 1
	CONFIGURE_COMMAND <SOURCE_DIR>/configure --prefix=${CMAKE_SOURCE_DIR} --shared
	BUILD_COMMAND make
	INSTALL_COMMAND make install
	)
